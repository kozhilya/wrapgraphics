\documentclass{article}

\begin{document}

\section*{Left vs.\ right wrapping}

\subsection*{Left (default)}
Text wraps around the left side of the image.

\wrapgraphics[scale=0.16, position=left]{img/tree.png}
\lipsum[1-3]

\newpage

\subsection*{Right}
Text wraps around the right side.

\wrapgraphics[scale=0.16, position=right]{img/tree.png}
\lipsum[4-6]

\newpage

\subsection*{Hang left}
Constant indent, left position.

\wrapgraphics[scale=0.16, position=left, hang=true]{img/tree.png}
\lipsum[7-9]

\newpage

\subsection*{Hang right}
Constant indent, right position.

\wrapgraphics[scale=0.16, position=right, hang=true]{img/tree.png}
\lipsum[10-12]
