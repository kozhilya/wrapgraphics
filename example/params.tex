\begin{document}

\section*{Parameter variations}

\subsection*{Low threshold (0.3)}
More of the image alpha is treated as opaque,
so the contour is closer to the image edge.

\wrapgraphics[threshold=0.3, padding=5, scale=0.20]{tree.png}
\lipsum[1-2]

\newpage

\subsection*{High threshold (0.8)}
Only the most opaque parts define the contour,
text can come closer.

\wrapgraphics[threshold=0.8, padding=5, scale=0.20]{tree.png}
\lipsum[1-2]

\newpage

\subsection*{Large padding (50px)}
Extra clearance between image edge and text.

\wrapgraphics[padding=50, scale=0.20]{tree.png}
\lipsum[1-2]

\newpage

\subsection*{No padding (0px)}
Text hugs the exact image contour.

\wrapgraphics[padding=0, scale=0.20]{tree.png}
\lipsum[1-3]
